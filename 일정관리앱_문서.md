## 1. 프로젝트 개요

단일 HTML 파일로 구성된 **주간·월간 일정 관리 앱**.  
Google 로그인 후 Firebase Firestore에 데이터를 저장해 PC·모바일 어디서든 자동 동기화된다.

### 핵심 특징
- 빌드 도구 없음 — HTML 파일 하나로 모든 것이 동작
- Firebase Firestore로 실시간 클라우드 동기화
- Google OAuth 로그인 (본인 계정 데이터만 접근 가능)
- 로컬(localhost / 파일 직접 열기) 환경에서는 로그인 없이 바로 사용 가능
- Soft Pastel 디자인 (Nunito 폰트, 요일별 파스텔 카드)
- 다크모드 지원

#### 8-4. Access Token 관리

**핵심 제약**: Firebase Auth `signInWithPopup`으로 받은 Google Access Token은 **1시간 후 만료**  
클라이언트에서 자동 갱신(refresh_token) 불가 → **수동 동기화 버튼 방식** 채택

```javascript
let _gcalToken = null;
let _gcalTokenExpiry = null;

// 로그인 시 토큰 저장
async function signInAndGetToken() {
  const provider = new firebase.auth.GoogleAuthProvider();
  provider.addScope('https://www.googleapis.com/auth/calendar');
  const result = await auth.signInWithPopup(provider);
  _gcalToken = result.credential.accessToken;
  _gcalTokenExpiry = Date.now() + 55 * 60 * 1000; // 55분 (5분 여유)
}

// 토큰 유효성 확인 — 만료면 재인증 팝업
async function getValidToken() {
  if (_gcalToken && Date.now() < _gcalTokenExpiry) return _gcalToken;
  await signInAndGetToken(); // 재인증 팝업
  return _gcalToken;
}
```

#### 8-6. PULL — Google Calendar → 이 앱

```javascript
async function pullFromGCal() {
  const token = await getValidToken();
  const now = new Date();
  const timeMin = new Date(now.getFullYear(), now.getMonth() - 1, 1).toISOString();
  const timeMax = new Date(now.getFullYear(), now.getMonth() + 3, 0).toISOString();

  const res = await fetch(
    `https://www.googleapis.com/calendar/v3/calendars/primary/events` +
    `?timeMin=${timeMin}&timeMax=${timeMax}&singleEvents=true&maxResults=500`,
    { headers: { Authorization: `Bearer ${token}` } }
  );
  const data = await res.json();
  const gcalEvents = data.items || [];

  let localItems = loadItems();

  for (const ge of gcalEvents) {
    if (ge.status === 'cancelled') {
      // 삭제된 이벤트 — 로컬에서도 제거
      localItems = localItems.filter(it => it.gcalId !== ge.id);
      continue;
    }

    const existing = localItems.find(it => it.gcalId === ge.id);
    const gcalUpdated = ge.updated;
    const mapped = mapGCalToLocal(ge); // 아래 참조

    if (!existing) {
      // 신규: 로컬에 추가
      localItems.push({ ...mapped, id: genId(), gcalId: ge.id });
    } else if (gcalUpdated > (existing.updatedAt || '')) {
      // 구글이 더 최신 → 덮어쓰기
      Object.assign(existing, mapped, { gcalId: ge.id });
    }
    // 로컬이 더 최신이면 무시 (PUSH 때 구글에 반영됨)
  }

  await saveItems(localItems);
}

function mapGCalToLocal(ge) {
  const isAllDay = !!ge.start?.date;
  return {
    type: 'event',
    title: ge.summary || '(제목 없음)',
    date: isAllDay ? ge.start.date : ge.start.dateTime?.slice(0, 10),
    start: isAllDay ? '' : ge.start.dateTime?.slice(11, 16),
    end:   isAllDay ? '' : ge.end?.dateTime?.slice(11, 16),
    memo: ge.description || '',
    color: 'yellow',
    category: '일상',
    repeat: 'none',
    repeatEnd: '',
    updatedAt: ge.updated,
  };
}
```

#### 8-7. PUSH — 이 앱 → Google Calendar

```javascript
async function pushToGCal() {
  const token = await getValidToken();
  const localItems = loadItems().filter(it => it.type === 'event');

  for (const it of localItems) {
    const body = mapLocalToGCal(it);

    if (it.gcalId) {
      // 기존 이벤트 수정 (PATCH)
      await fetch(
        `https://www.googleapis.com/calendar/v3/calendars/primary/events/${it.gcalId}`,
        { method: 'PATCH', headers: { Authorization: `Bearer ${token}`, 'Content-Type': 'application/json' },
          body: JSON.stringify(body) }
      );
    } else {
      // 신규 이벤트 생성 (POST)
      const res = await fetch(
        `https://www.googleapis.com/calendar/v3/calendars/primary/events`,
        { method: 'POST', headers: { Authorization: `Bearer ${token}`, 'Content-Type': 'application/json' },
          body: JSON.stringify(body) }
      );
      const created = await res.json();
      it.gcalId = created.id; // 로컬 항목에 gcalId 저장
    }
  }

  await saveItems(localItems); // gcalId 업데이트 반영
}

function mapLocalToGCal(it) {
  if (it.start) {
    return {
      summary: it.title,
      description: it.memo || '',
      start: { dateTime: `${it.date}T${it.start}:00`, timeZone: 'Asia/Seoul' },
      end:   { dateTime: `${it.date}T${(it.end || it.start)}:00`, timeZone: 'Asia/Seoul' },
    };
  }
  return {
    summary: it.title,
    description: it.memo || '',
    start: { date: it.date },
    end:   { date: it.date },
  };
}
```

```javascript
async function deleteItem() {
```
